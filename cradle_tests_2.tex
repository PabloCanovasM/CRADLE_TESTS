	\begin{frame}{Simulation Wu Experiment}
		Consider atoms of $\mathrm{^{60}Co}$ in a thermal bath and in the presence of a magnetic field in the -z direction. 
		
		Model each nuclei as independent 11 state system (each of the values of $m_j$)
		
		$$Z = \sum_{m_j = -5}^{5} e^{\frac{m_j \mu_N({\mathrm{^{60}Co}}) B}{5k_bT}} \rightarrow P(m_j) = \frac 1Ze^{\frac{m_j \mu_N({\mathrm{^{60}Co}}) B}{5k_bT}}$$
		
		From here polarisation and alignment in Z direction ($P_z$ and $\mathcal{A}_z$) can be computed
		
		$$ P_z = \frac{1}{J}\left<m_j\right> = \sum_{m_j = -5}^{5} \frac{m_jP(m_j)}{5}\qquad\mathcal{A}_z = \frac{3\left<m^2_j\right>-J(J+1)}{J(2J-1)}$$
		
		which leads to non-zero A polarisation.
	\end{frame}

	\begin{frame}{Gamow-Teller Decay: $\mathrm{^{60}Co}$}{Numerical evaluation}
		Use that distributions in $z_e$, $z_\nu$, $\cos \theta_{e,\nu} \equiv z_{e,\nu}$ and $\phi$ are known if $F \ge 0$ for all orientations of $\boldsymbol{p_e}, \boldsymbol{p_\nu}$ ($\boldsymbol{J}$ fixed).
		$$f_1(z_e) = \frac{1 + \left<b\gamma_e^{-1}\right> + \left<A\beta_e\right>z_e}{2(1 + \left<b\gamma_e^{-1}\right>)}$$
		$$f_2(z_\nu) = \frac{1 + \left<b\gamma_e^{-1}\right> + \left<B\right>z_\nu}{2(1 + \left<b\gamma_e^{-1}\right>)}$$
		$$f_3(z_{e,\nu}) = \frac{1 + \left<b\gamma_e^{-1}\right> + \left<a\beta_e\right>z_{e,\nu}}{2(1 + \left<b\gamma_e^{-1}\right>)}$$
		$$f_4(\phi) = \frac{1 + \left<b\gamma_e^{-1}\right> + \left<\left(a+\frac c3\right)\beta_e\right>\frac{\pi^2}{16}\cos\phi+\left<D\beta_e\right>\frac{\pi^2}{16}\sin\phi}{2\pi(1 + \left<b\gamma_e^{-1}\right>)}$$
		Averages computed numerically using f(E) from the simulation data itself (avoid computing the Fermi function myself)
	\end{frame}
	\begin{frame}{Gamow-Teller Decay: $\mathrm{^{60}Co}$}{Numerical verification}
		From distributions:
		\begin{itemize}
			\item Compute difference $\Delta_{i,j} = f_j(x_i)-f_{j,th}(x_i)$ for each point $x_i$ in each distribution $f_j$
			\item Use $\sigma_{i,j} = \sqrt{f_j(x_i)}$ as uncertainity
			\item Compute residuals as:
			$$Res_{i,j} = \frac{\Delta_{i,j}}{\sigma_{i,j}} \rightarrow \chi^2_j = \sum_i Res^2_{i,j} $$
			\item Verify $\chi^2_j \approx \#\{x_i\}$ and residuals mostly between -2 and 2.
		\end{itemize}
	\end{frame}

\begin{frame}{Plausible scenarios with non-zero D}
	Looking at D in terms of $C_X$,$C'_X$	
	$$D = 2\delta_{J_i,J_f}M_FM_{GT}\sqrt{\frac{J_i}{J_i+1}} \left[\vphantom{\frac{\alpha}{p_e}} Im\left(C_S{C_T}^\ast+C'_S{C'_T}^\ast-C_V{C_A}^\ast-C'_V{C'_A}^\ast\right)\right.$$$$\left.\mp\frac{\alpha Z m_e}{p_e}Re\left(C_S{C_A}^\ast+C_S{C'_A}^\ast+C_T{C_V}^\ast+C'_T{C'_V}^\ast\right)\right]$$
	
	Multiple simple options:
	\begin{itemize}
		\item $C_A=C_A'$ with an imaginary component, $C_V=C'_V=1$
		\item Keep $C^{(\prime)}_A$ and $C^{(\prime)}_V$, and add either a $C_S = C'_S$ or a $C_T= C'_T$ coupling with a real component (simpler $C_S$)
		\item Keep $C^{(\prime)}_A$ and $C^{(\prime)}_V$, add add $C_S=-C'_S$ and $C_T=-C'_T$ couplings, one real and other imaginary
		\item Keep $C^{(\prime)}_A$ and $C^{(\prime)}_V$, and add $C_S=C'_S$ and $C_T=C'_T$ couplings, one real and other imaginary (not so simple)
	\end{itemize}
\end{frame}
\begin{frame}{Imaginary $C_VC_A$}
	Idea: add just a phase shift between  $C_A$ and $C_V$, value $\theta$.
	\begin{itemize}
		\item Global symmetry to a phase shift of all coupling constants $\rightarrow$ $C_V$ chosen 1
		\item $|C_A|$ untouched $\rightarrow$ $\xi$,$a$,$b=0$,$c$ constant
		\item $\delta D \propto \sin(\theta) \approx \theta$ and $\delta D \propto \left<\beta_e\right>  (good)$
		\item $\delta A ,\delta B \propto 1-\cos(\theta) = O(\theta^2)$
	\end{itemize}
	\small$$A\xi = 2|M_{GT}|^2 \lambda_{JJ'} \left[ \pm  Re \left(C_T {C'_T}^\ast - C_A {C'_A}^*\right) + \frac{\alpha Z m_e}{p_e} Im \left(C_T {C'_A}^* + C'_T {C_A}^*\right) \right] $$
	\small$$+ 2\delta_{J'J} M_F M_{GT} \sqrt{\frac{J}{J+1}} \left[ Re (C_S C'_T{}^* + C'_S C_T{}^* - C_V C'_A{}^* - C'_VC_A{}^*)\vphantom{\frac{a}{b}}\right. $$
	\small$$\left.\pm \frac{\alpha Z m}{p_e} Im \left(C_S C'_A{}^* + C'_S C_A{}^* - C_V C'_T{}^* - C'_V C_T{}^* \right)\right]$$
\end{frame}

\begin{frame}{Imaginary $C_VC_A$}
	For theory: using $|C_V| = |C_A| = 1$ and $\theta$ the angle between $C_V$ and $C_A$ 
	
	$$D\left<\beta_e\right> = \frac{4M_{GT}\sin\theta\sqrt{\frac{2}{3}}}{2+2|M_{GT}|^2}\left<\beta_e\right>$$
	
	For experiment: noting $b = 0$, we can relate $N (\phi > 0)$ and   $N (\phi < 0)$ to the distribution
	
	\small $$\frac{N(\phi > 0)}{N} = \int_{0}^{\pi} f_4(\phi) = \int_{0}^{\pi}  \frac{1  + \left<\left(a+\frac c3\right)\beta_e\right>\frac{\pi^2}{16}\cos\phi+\left<D\beta_e\right>\frac{\pi^2}{16}\sin\phi}{2\pi} $$
	$$  \frac{N(\phi > 0)}{N} = \frac{1}{2} + D\left<\beta_e\right>\frac{\pi}{16} \rightarrow D\left<\beta_e\right> = \frac {N(\phi > 0) - N(\phi < 0)}{N (\phi > 0)+N (\phi < 0)}\frac{8}{\pi}$$
\end{frame}
