	\begin{frame}{Simulation Wu Experiment}
		Consider atoms of $\mathrm{^{60}Co}$ in a thermal bath and in the presence of a magnetic field in the -z direction. 
		
		Model each nuclei as independent 11 state system (each of the values of $m_j$)
		
		$$Z = \sum_{m_j = -5}^{5} e^{\frac{m_j \mu_N({\mathrm{^{60}Co}}) B}{5k_bT}} \rightarrow P(m_j) = \frac 1Ze^{\frac{m_j \mu_N({\mathrm{^{60}Co}}) B}{5k_bT}}$$
		
		From here polarisation and alignment in Z direction ($P_z$ and $\mathcal{A}_z$) can be computed
		
		$$ P_z = \frac{1}{J}\left<m_j\right> = \sum_{m_j = -5}^{5} \frac{m_jP(m_j)}{5}\qquad\mathcal{A_z} = \frac{3\left<m^2_j\right>-J(J+1)}{J(2J-1)}$$
		
		which leads to non-zero A polarisation.
	\end{frame}

	\begin{frame}{Gamow-Teller Decay: $\mathrm{^{60}Co}$}{Numerical evaluation}
		Use that distributions in $z_e$, $z_\nu$, $\cos \theta_{e,\nu} \equiv z_{e,\nu}$ and $\phi$ are known if $F \ge 0$ for all orientations of $p\boldsymbol{p_e}, \boldsymbol{p_\nu}$ ($\boldsymbol{J}$ fixed).
		$$f_1(z_e) = \frac{1 + \left<b\gamma_e^{-1}\right> + \left<A\beta_e\right>z_e}{2(1 + \left<b\gamma_e^{-1}\right>)}$$
		$$f_2(z_\nu) = \frac{1 + \left<b\gamma_e^{-1}\right> + \left<B\right>z_\nu}{2(1 + \left<b\gamma_e^{-1}\right>)}$$
		$$f_3(z_{e,\nu}) = \frac{1 + \left<b\gamma_e^{-1}\right> + \left<a\beta_e\right>z_{e,\nu}}{2(1 + \left<b\gamma_e^{-1}\right>)}$$
		$$f_4(\phi) = \frac{1 + \left<b\gamma_e^{-1}\right> + \left<\left(a+\frac c3\right)\beta_e\right>\frac{\pi^2}{16}\cos\phi+\left<D\beta_e\right>\frac{\pi^2}{16}\sin\phi}{2\pi(1 + \left<b\gamma_e^{-1}\right>)}$$
		Averages computed numerically using f(E) from the simulation data itself (avoid computing the Fermi function myself)
	\end{frame}
	\begin{frame}{Gamow-Teller Decay: $\mathrm{^{60}Co}$}{Numerical verification}
		From distributions:
		\begin{itemize}
			\item Compute difference $\Delta_{i,j} = f_j(x_i)-f_{j,th}(x_i)$ for each point $x_i$ in each distribution $f_j$
			\item Use $\sigma_{i,j} = \sqrt{f_j(x_i)}$ as uncertainity
			\item Compute residuals as:
			$$Res_{i,j} = \frac{\Delta_{i,j}}{\sigma_{i,j}} \rightarrow \chi^2_j = \sum_i Res^2_{i,j} $$
			\item Verify $\chi^2_j \approx \#\{x_i\}$ and residuals mostly between -2 and 2.
		\end{itemize}
	\end{frame}
