				\item $C_T = C'_T$ pure real (and large)
				\item $C_T = C'_T$ pure imaginary 
				\item $C_T = -C'_T$, either real or imaginary
			\end{itemize}

\begin{frame}{Gamow-Teller Decay: $\mathrm{^{60}Co}$}{Numerical evaluation}
	Use that distributions in $z_e$, $z_\nu$, $\cos \theta_{e,\nu} \equiv z_{e,\nu}$ and $\phi$ are known if $F \ge 0$ for all orientations of $p\boldsymbol{p_e}, \boldsymbol{p_\nu}$ ($\boldsymbol{J}$ fixed).
	$$f(z_e) = \frac{1 + \left<b\gamma_e^{-1}\right> + \left<A\beta_e\right>z_e}{2(1 + \left<b\gamma_e^{-1}\right>)}$$
	$$f(z_\nu) = \frac{1 + \left<b\gamma_e^{-1}\right> + \left<B\right>z_\nu}{2(1 + \left<b\gamma_e^{-1}\right>)}$$
	$$f(z_{e,\nu}) = \frac{1 + \left<b\gamma_e^{-1}\right> + \left<a\beta_e\right>z_{e,\nu}}{2(1 + \left<b\gamma_e^{-1}\right>)}$$
	$$f(\phi) = \frac{1 + \left<b\gamma_e^{-1}\right> + \left<\left(a+\frac c3\right)\beta_e\right>\frac{\pi^2}{16}\cos\phi+\left<D\beta_e\right>\frac{\pi^2}{16}\sin\phi}{2\pi(1 + \left<b\gamma_e^{-1}\right>)}$$
	Averages computed numerically using f(E) from the simulation data itself (avoid computing the Fermi function myself)
\end{frame}
