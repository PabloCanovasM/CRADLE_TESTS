\begin{frame}{Gamow-Teller Decay: $\mathrm{^{60}Co}$}
	\begin{columns}
		\begin{column}{0.575\textwidth}
			Properties of $\mathrm{^{60}Co} (5^+) \rightarrow \mathrm{^{60}Ni} (4^+)$
			\begin{itemize}
				\item Q = 317.06 keV (not fully ideal)
				\item $J_f = J_i - 1 \rightarrow \lambda_{J_i,J_f} = \Lambda_{J_i,J_f} = 1$ 
				\item 2 $\gamma$ almost always ($5^+\rightarrow2^+$ only 1 $\gamma$)
			\end{itemize}
			
			Many cases to consider, though for realism: keep $C_A = C'_A = \mathrm{cte}( = 1)$.
			\begin{itemize}
				\item $C_T = C'_T = 0$ (Standard Model)
				\item $C_T = C'_T$ pure real (and large)
				\item $C_T = C'_T$ pure imaginary 
				\item $C_T = -C'_T$, either real or imaginary
			\end{itemize}	
		\end{column}
		\begin{column}{0.425\textwidth}

\begin{frame}{Gamow-Teller Decay: $\mathrm{^{60}Co}$}{Numerical evaluation}
	Use that distributions in $z_e$, $z_\nu$, $\cos \theta_{e,\nu} \equiv z_{e,\nu}$ and $\phi$ are known.
	$$f(z_e) = \frac{1 + \left<b\gamma_e^{-1}\right> + \left<A\beta_e\right>z_e}{2(1 + \left<b\gamma_e^{-1}\right>)}$$
	$$f(z_\nu) = \frac{1 + \left<b\gamma_e^{-1}\right> + \left<B\beta_e\right>z_\nu}{2(1 + \left<b\gamma_e^{-1}\right>)}$$
	$$f(z_{e,\nu}) = \frac{1 + \left<b\gamma_e^{-1}\right> + \left<a\beta_e\right>z_{e,\nu}}{2(1 + \left<b\gamma_e^{-1}\right>)}$$
	$$f(\phi) = \frac{1 + \left<b\gamma_e^{-1}\right> + \left<\left(a+\frac c3\right)\beta_e\right>\frac{\pi^2}{16}\cos\phi+\left<D\beta_e\right>\frac{\pi^2}{16}\sin\phi}{2\pi(1 + \left<b\gamma_e^{-1}\right>)}$$
	Averages computed numerically using f(E) from the simulation data itself (avoid computing the Fermi function myself)
\end{frame}
